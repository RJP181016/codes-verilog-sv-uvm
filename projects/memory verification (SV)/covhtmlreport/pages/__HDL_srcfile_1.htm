1  module memory(clk,rst,wt_rd,addr,wdata,rdata,valid,ready);
     2  parameter WIDTH=16;
     3  parameter DEPTH=64;
     4  parameter ADDR_WIDTH=$clog2(DEPTH);
     5  
     6  input clk,rst;
     7  input wt_rd;
     8  input [ADDR_WIDTH-1:0]addr;
     9  input [WIDTH-1:0]wdata;
     10  output reg [WIDTH-1:0]rdata;
     11  input valid;
     12  output reg ready;
     13  integer i;
     14  
     15  reg [WIDTH-1:0] mem [DEPTH-1:0];
     16  
     17  always@(posedge clk) begin
     18  	if(rst) begin
     19  		rdata=0;
     20  		ready=0;
     21  		for(i=0;i<DEPTH;i=i+1) begin
     22  			mem[i]=0;
     23  		end
     24  	end
     25  	else begin
     26  		if(valid) begin
     27  			ready=1;
     28  			if(wt_rd) begin
     29  				mem[addr]=wdata;
     30  			end
     31  			else begin
     32  				rdata=mem[addr];
     33  			end
     34  		end
     35  		else begin
     36  			ready=0;
     37  		end
     38  	end
     39  end
     40  
     41  endmodule
