1  class mem_gen;

     6  
     7  function new(int i);

     12  case(mem_common::testname)
     13  "test_1_wt_1_rd": begin
     14  	//write a location

     18  	mem_common::gen2bfmDA[gen_num].put(tx);

     32  		mem_common::gen2bfmDA[gen_num].put(tx);
     33  	end

     35  		//read a location
     36  		tx=new();
     37  		assert(tx.randomize() with {wt_rd==0;});
     38  		tx.addr=addr_locQ.pop_front();
     39  		//$display("popped addr: tx.addr=%0d",tx.addr);
     40  		mem_common::gen2bfmDA[gen_num].put(tx);
     41  	end
     42  end
